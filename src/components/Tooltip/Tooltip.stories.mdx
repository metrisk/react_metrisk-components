# Tooltip
An inline hyperlink using React Router.
 
## Base
For use within paragraphs or other instances where a button would not be visually appropriate.

<Preview>
  <Story name='Base'>
    {() => {
      const [el, setEl] = useState()
      const ref = useRef()
      useEffect(() => {
        setEl(ref.current)
      }, [])
      return (
        <Text>
          Lorem ipsum dolor sit amet, consectetur adipiscing elit, <strong ref={ref} style={{ display: 'inline-block'}}>toggle</strong> do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat.
          
          <Tooltip attachTo={el} align={'top'}>
            I am a Tooltip!
          </Tooltip>
        </Text>
      )
    }}
  </Story>
</Preview>
